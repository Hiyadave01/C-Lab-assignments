Q1) WAP TO ENTER NUMBER TILL THE USER WANTS. AT THE END IT SHUD DISPLAY THE COUNT OF POSITIVE, NEGATIVE, AND ZEROES ENTERED.

#include<stdio.h>

int main(){
    int num;
    int negativecount = 0, positivecount = 0, zerocount = 0;
    char ch;

    do{
        printf("Enter a number: ");
        scanf("%d", &num);

        if(num>0){
            positivecount++;
        }
        else if(num<0){
            negativecount++;
        }
        else{
            zerocount++;
        }
        
        //Asking user if they want to continue
        printf("Do you want to continue (y/n)? ");
        scanf(" %c", &ch); 

        }
        while(ch == 'y' || ch == 'Y');

        printf("Positive numbers count: %d\n", positivecount);
        printf("Negative numbers count: %d\n", negativecount);
        printf("Zero count: %d\n", zerocount);
    return 0;
        }
    

Q2) WAP TO PRINT THE MULTIPLICATION TABLE OF THE NUMBER ENTERED BY THE USER. IT SHUD BE IN THE CORRECT FORMATTING. 

#include <stdio.h>
int main()
{
    int num, i;

    // Taking user input
    printf("Enter a number to display its multiplication table: ");
    scanf("%d", &num);

    printf("Multiplication table of %d:\n", num);
    for (i = 1; i <= 10; i++)
    {
        printf("%d * %d = %d\n", num, i, num * i);
    }
    return 0;
}

Q3 . A) PYRAMID

#include<stdio.h>
int main()
{
    int i, j, k, rows;
    printf("Enter the number of rows: ");
    scanf("%d", &rows);

    for(i=1; i<=rows; i++)
    {
        for(j=i; j<rows; j++)
        {
            printf(" ");    
        }
        for(k=1; k<=i; k++)
        {
            printf("* ");
        }
        printf("\n");
    }
    return 0;  
    
}

Q4. ) POPULATION INC

#include <stdio.h>
int main(void) 
{
    double population = 100000.0;  
    double growth = 0.10;      // 10% per year expressed as decimal (10% = 0.10)
    int n = 10;                                    
    for (int i = 1; i <= n; ++i) 
    {
        population = population * (1.0 + growth);
        printf("End of year %d: population = %.2f\n", i, population);
    }
    return 0;
}
