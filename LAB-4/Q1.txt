Q1 ) CHECK IF TRIANGLE IS SCALENE EQUILATERAL ISOSCELES OR RIGHT ANGLED

#include<stdio.h>

int main(){

    float a, b, c;

    printf("Enter the lengths of three side of the triangle:");
    scanf("%f %f %f", &a, &b, &c);

    if(a==b && b==c)
        {
            printf("The triangle is equilateral.\n");
        }
    else if(a==b || b==c || a==c)
    {
        printf("The triangle is isosceles.\n");
    }
    else if( a*a == b*b + c*c || b*b == a*a + c*c || c*c == a*a + b*b)
    {
        printf("The triangle is right angled.\n");
    }
    else {
        printf("The triangle is scalene.\n");
    }
    return 0;
}
