
Q3.B ) RIGHT ANGLED TRIANGLE

#include<stdio.h>
int main()
{
    int i, j , num = 1, rows;
    printf("Enter the number of rows: ");
    scanf("%d", &rows);

    for(i=1; i<=rows; i++)
    {
        for(j=1; j<=i; j++)
        {
            printf("%d ",num);
            num++;
        }
        printf("\n");
    }
    return 0;
}
