Q4. ) POPULATION INC

#include <stdio.h>
int main(void) 
{
    double population = 100000.0;  
    double growth = 0.10;      // 10% per year expressed as decimal (10% = 0.10)
    int n = 10;                                    
    for (int i = 1; i <= n; ++i) 
    {
        population = population * (1.0 + growth);
        printf("End of year %d: population = %.2f\n", i, population);
    }
    return 0;
}
