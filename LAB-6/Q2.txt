// Count positive, negative, odd, even elements in an array.
#include <stdio.h>
int main()
{
    int n, positive_count = 0, negative_count = 0, even = 0, odd = 0;
    printf("Enter size of array: ");
    scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++)
    {
        printf("Enter value number %d ", i + 1);
        scanf("%d", &arr[i]);
        if (arr[i] > 0)
        {
            positive_count++;
        }
        else if (arr[i] < 0)
        {
            negative_count++;
        }
        else if (arr[i] % 2 == 0)
        {
            even++;
        }
        else
        {
            odd++;
        }
    }
    printf("Positive elements: %d\n", positive_count);
    printf("Negative elements: %d\n", negative_count);
    printf("Even: %d\n", even);
    printf("Odd: %d\n", odd);
    return 0;
}
