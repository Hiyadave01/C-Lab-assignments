//Count positive, negative, and zero elements in an array.
#include <stdio.h>
int main()
{
    int n,positive_count=0, negative_count=0, zero_count=0;
    printf("Enter size of array: ");
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter value number %d ",i+1);
        scanf("%d",&arr[i]);
        if(arr[i]>0)
        {
           positive_count++;
        }
        else if(arr[i]<0)
        {
           negative_count++;
        }
        else
        {
           zero_count++;
        }
    }
    printf("Positive elements: %d\n",positive_count);
    printf("Negative elements: %d\n",negative_count);
    printf("Zeroes: %d\n",zero_count);
    return 0;

}
