//Find the second largest element in an array.

#include <stdio.h>
int main()
{
    int a;

    printf("Enter the numbers of integers:");
    scanf("%d", &a);

    if (a < 2)
    {
        printf("The input is invalid.");
        return 0;
    }

    int arr[a];
    printf("Enter %d integers\n", a);
    for (int i = 0; i < a; i++)
    {
        scanf("%d", &arr[i]);
    }
    int largest, second_largest;
    if (arr[0] > arr[1])
    {
        largest = arr[0];
        second_largest = arr[1];
    }
    else
    {
        largest = arr[1];
        second_largest = arr[0];
    }

    for (int i = 0; i < a; i++) // loop for second largest
    {
        if (arr[i] > largest)
        {
            second_largest = largest;
            largest = arr[i];
        }
        else if (arr[i] > second_largest && arr[i] != largest)
        {
            second_largest = arr[i];
        }
    }

    if (largest == second_largest) // all elements equal
    {
        printf("No second largest element possible.");
        return 0;
    }
    else
    {
        printf("The second Largest integer is:%d\n", second_largest);
    }
    return 0;
}
