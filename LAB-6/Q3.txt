//WAP TO READ A LIST OF INTEGERS AND STORE IT IN A SINGLE DIMENSION ARRAY. WRITE A C PROGRAM TO FIND THE FREQUENCY OF A PARTICULAR NUMBER IN A LIST OF INTEGERS

#include <stdio.h>
int main()
{
    int n,num,c=0;
    printf("Enter size of array: ");
    scanf("%d",&n);
    int arr[n];
    printf("Enter %d elements ",n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    
    printf("Enter number you want to check frequency of: ");
    scanf("%d",&num);
    for(int i=0;i<n;i++)
    {
        if(arr[i]==num)
        {
           c++;
        }
    }

    printf("The given number occurs %d times.",c);
    return 0;

}
