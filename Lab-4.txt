Q1 ) CHECK IF TRIANGLE IS SCALENE EQUILATERAL ISOSCELES OR RIGHT ANGLED

#include<stdio.h>

int main(){

    float a, b, c;

    printf("Enter the lengths of three side of the triangle:");
    scanf("%f %f %f", &a, &b, &c);

    if(a==b && b==c)
        {
            printf("The triangle is equilateral.\n");
        }
    else if(a==b || b==c || a==c)
    {
        printf("The triangle is isosceles.\n");
    }
    else if( a*a == b*b + c*c || b*b == a*a + c*c || c*c == a*a + b*b)
    {
        printf("The triangle is right angled.\n");
    }
    else {
        printf("The triangle is scalene.\n");
    }
    return 0;
}


Q2 ) BMI 
 
#include<stdio.h>
int main()
{
    float BMI, weight, height, N1, N2;
    printf("Enter weight in kg:");
    scanf("%f", &weight);
    printf("Enter height in meters:");
    scanf("%f", &height);
    printf("Enter the value of N1 N2:");
    scanf("%f %F", &N1, &N2);
    
    BMI = weight / height*height;
    printf("Your BMI is: %.2f\n",BMI);


    //Validate input
    if(height<=0 && weight< N1 || height<=0 && weight > N2){
        printf("Invalid input.\n");
        return 0;
    }
    else
       {printf("You are classified as: ");}

    //BMI Categories
    if(BMI<15){
        printf("STARVATION.\n");
    }
    else if(BMI >= 15.1 && BMI <= 17.5){
        printf("ANOREXIC.\n");
    }
    else if(BMI >= 17.6 && BMI <= 18.5){
        printf("UNDERWEIGHT.\n");
    }  
    else if(BMI >= 18.6 && BMI <= 24.9){
        printf("IDEAL.\n");
    }
    else if(BMI >= 25 && BMI <= 29.9){
        printf("OVERWEIGHT.\n");
    }
    else if(BMI >= 30 && BMI <= 39.9){
        printf("OBESE.\n");
    }
    else{
        printf("MORBIDLY OBESE.\n");
    }
    return 0;
}


Q3 ) COLLINEAR POINTS 

#include<stdio.h>
int main(){
    float x1, y1, x2, y2, x3, y3, Area;
    printf("Enter coordinates of first point (x1 y1): ");
    scanf("%f %f", &x1, &y1);

    printf("Enter coordinates of second point (x2 y2): ");
    scanf("%f %f", &x2, &y2);

    printf("Enter coordinates of third point (x3 y3): ");
    scanf("%f %f", &x3, &y3);

    Area =  ( x1 * (y2 - y3) + x2 * (y3 - y1) + x3 * (y1 - y2)  / 2 );

    if(Area == 0)
    {
        printf("The points are collinear.\n");
    }
    else
    {
        printf("The points are not collinear.\n");
    }
  return 0;

}
